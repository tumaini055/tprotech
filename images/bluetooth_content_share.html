    <h1>Welcome to Our Online Store</h1>
    <div id="products">
        <!-- List of products with buttons to add to cart -->
        <div>
            <h2>Product 1</h2>
            <p>Price: $10</p>
            <button onclick="addToCart('Product 1', 10)">Add to Cart</button>
        </div>
        <!-- Add more products here -->
    </div>

    <h2>Shopping Cart</h2>
    <div id="cart">
        <!-- Cart content will be displayed here -->
    </div>

    <script>
        let cart = [];

        function addToCart(productName, price) {
            cart.push({ productName, price });
            updateCart();
        }

        function updateCart() {
            let cartDiv = document.getElementById("cart");
            cartDiv.innerHTML = "<h3>Cart:</h3>";
            let total = 0;

            for (let item of cart) {
                cartDiv.innerHTML += `<p>${item.productName} - $${item.price}</p>`;
                total += item.price;
            }

            cartDiv.innerHTML += `<p>Total: $${total}</p>`;
        }
    </script>
